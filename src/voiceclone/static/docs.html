Synthesize Speech

POST /api/v1/tts/synthesize

Convert text to speech using a cloned voice. Returns a base64 encoded audio URL.

Request Body

Parameter Type Required Description
text string Yes Text to convert to speech (1-5000 characters)
voice_id UUID Yes ID of the cloned voice to use
model string No TTS model: svara, xtts, chatterbox, orpheus. Default: svara
language string No Language code. Default: hi
emotion string No Emotion: happy, sad, anger, fear, neutral
speaker_gender string No male or female. Default: female
output_format string No wav or mp3. Default: wav

Example Request

Python 
import requests

response = requests.post(
    "https://voiceclone.gahfaudio.in/api/v1/tts/synthesize",
    json={
        "text": "नमस्ते, मैं आपकी कैसे मदद कर सकता हूं?",
        "voice_id": "your-voice-uuid",
        "model": "svara",
        "language": "hi",
        "emotion": "happy",
        "speaker_gender": "female"
    }
)

result = response.json()
print(result["audio_url"])  # data:audio/wav;base64,...

Synthesize Audio (Direct Download)

POST /api/v1/tts/synthesize/audio

Same as synthesize, but returns the audio file directly as a streaming response. Ideal for direct playback or saving to file.

Example Request

Python 
import requests

response = requests.post(
    "https://voiceclone.gahfaudio.in/api/v1/tts/synthesize/audio",
    json={
        "text": "Hello, how can I help you today?",
        "voice_id": "your-voice-uuid",
        "model": "svara",
        "language": "en-in"
    }
)

# Save audio file directly
with open("output.wav", "wb") as f:
    f.write(response.content)

# Get metadata from headers
duration = response.headers.get("X-Duration-Seconds")
print(f"Duration: {duration}s")

Voice Management

POST /api/v1/voices

Create a new voice profile by uploading a reference audio sample.

Request (multipart/form-data)

Parameter Type Description
name string Name for the voice profile
audio_file file Reference audio (WAV, MP3, FLAC, OGG, M4A). 10-30 seconds recommended.
description string Optional description
Python 
import requests

with open("voice_sample.wav", "rb") as f:
    response = requests.post(
        "https://voiceclone.gahfaudio.in/api/v1/voices",
        files={"audio_file": f},
        data={
            "name": "My Sales Agent Voice",
            "description": "Professional Hindi voice"
        }
    )

voice = response.json()
print(f"Voice ID: {voice['id']}")

WebSocket Streaming API

WebSocket wss://voiceclone.gahfaudio.in/ws/tts

Real-time streaming TTS for low-latency applications like live conversations and AI agents.

Connection Flow

1. Connect to WebSocket endpoint
2. Send JSON request with text and voice settings
3. Receive audio chunks in real-time
4. Receive final message with total duration

Python Example

Python 
import asyncio
import websockets
import json
import base64

async def stream_tts():
    uri = "wss://voiceclone.gahfaudio.in/ws/tts"

    async with websockets.connect(uri) as ws:
        # Send synthesis request
        request = {
            "text": "नमस्ते, यह एक टेस्ट मैसेज है।",
            "voice_id": "your-voice-uuid",
            "model": "svara",
            "language": "hi"
        }
        await ws.send(json.dumps(request))

        audio_chunks = []

        while True:
            message = json.loads(await ws.recv())

            if message.get("type") == "chunk":
                # Decode and store audio chunk
                chunk = base64.b64decode(message["audio_base64"])
                audio_chunks.append(chunk)
                print(f"Received chunk {message['chunk_index']}")

            elif message.get("type") == "end":
                print(f"Complete! Duration: {message['total_duration_seconds']}s")
                break

            elif message.get("type") == "error":
                print(f"Error: {message['error']}")
                break

        # Combine and save audio
        full_audio = b"".join(audio_chunks)
        with open("stream_output.wav", "wb") as f:
            f.write(full_audio)

asyncio.run(stream_tts())

JavaScript Example

JavaScript 
const ws = new WebSocket("wss://voiceclone.gahfaudio.in/ws/tts");

const audioChunks = [];
const audioContext = new (window.AudioContext || window.webkitAudioContext)();

ws.onopen = () => {
  ws.send(JSON.stringify({
    text: "Hello, this is a streaming test.",
    voice_id: "your-voice-uuid",
    model: "svara",
    language: "en-in"
  }));
};

ws.onmessage = async (event) => {
  const message = JSON.parse(event.data);

  if (message.type === "chunk") {
    // Decode base64 audio chunk
    const binaryString = atob(message.audio_base64);
    const bytes = new Uint8Array(binaryString.length);
    for (let i = 0; i < binaryString.length; i++) {
      bytes[i] = binaryString.charCodeAt(i);
    }
    audioChunks.push(bytes);

  } else if (message.type === "end") {
    console.log(`Complete! Duration: ${message.total_duration_seconds}s`);
    // Play the audio
    playAudio(audioChunks);
  }
};

function playAudio(chunks) {
  const blob = new Blob(chunks, { type: "audio/wav" });
  const audio = new Audio(URL.createObjectURL(blob));
  audio.play();
}

AI Sales Agent Integration

Complete example for integrating VoiceClone TTS with your AI sales agent platform.

Python 
import httpx
import base64

class VoiceCloneClient:
    """VoiceClone TTS client for AI sales agents."""

    def __init__(self, voice_id: str, model: str = "svara"):
        self.base_url = "https://voiceclone.gahfaudio.in/api/v1"
        self.voice_id = voice_id
        self.model = model
        self.client = httpx.AsyncClient(timeout=60.0)

    # Map AI sentiment to TTS emotion
    EMOTION_MAP = {
        "positive": "happy",
        "negative": "sad",
        "urgent": "fear",
        "frustrated": "anger",
        "default": "neutral"
    }

    async def speak(
        self,
        text: str,
        language: str = "hi",
        sentiment: str = "default"
    ) -> bytes:
        """Convert text to speech with emotion based on sentiment."""
        emotion = self.EMOTION_MAP.get(sentiment, "neutral")

        response = await self.client.post(
            f"{self.base_url}/tts/synthesize/audio",
            json={
                "text": text,
                "voice_id": self.voice_id,
                "model": self.model,
                "language": language,
                "emotion": emotion
            }
        )
        response.raise_for_status()
        return response.content

Error Handling

Status Code Error Description
400 Bad Request Invalid parameters or voice not ready
404 Not Found Voice ID not found
413 Payload Too Large Text exceeds 5000 characters or audio file too large
429 Too Many Requests Rate limit exceeded (100 requests/minute)
502 Bad Gateway TTS service error (Modal.com inference)

Error Response Format

{
  "detail": "Voice not found: 123e4567-e89b-12d3-a456-426614174000"
